224 public static void AddCodeBlockToOB(XmlLogicalBlock codeBlock, ResourceManagement resourceManagement, TiaTemplateContext tiaTemplateContext, PlcDevice plcDevice)
225 {
226 var existingOB = OpennessFuncs.TryGetOBFromPlcDevice(plcDevice, codeBlock.Name);
227 //Create new OB in SCL with calling code
228 if (existingOB == null)
229 {
230 if (codeBlock.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
231 {
232 codeBlock.GenerateSclBlock(plcDevice);
233 }
234 else //insert networks to KOP/FUP
235 {
236 codeBlock.GenerateXmlBlock(plcDevice, null, tiaTemplateContext.TiaProject.GetEditingLanguage());
237 }
238 }
239 else
240 //Insert Code to existing OB
241 {
242 var blockAsDoc = OpennessFuncs.ExportBlockAsXml(existingOB.Name, plcDevice, OpennessFuncs.CachingOptions.DISABLED);
243 Parser parser = new Parser();
244 var xmlOB = parser.ParseXml(blockAsDoc);
245 xmlOB.MergeBlock(codeBlock);
246 //add SCL-Code
247 if (xmlOB.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
248 {
249 xmlOB.GenerateSclBlock(plcDevice);
250 }
251 else //insert networks to KOP/FUP
252 {
253 xmlOB.GenerateXmlBlock(plcDevice, null, tiaTemplateContext.TiaProject.GetEditingLanguage());
254 }
255 }
